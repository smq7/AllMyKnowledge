git init -- ініціалізація папки за якою буде слідкувати git
git add -- додавання файлів які мають бути закоміченні 
git commit -- коміт файлів 
git log -- переглядання історії коммітів
git diff -- показує які були змінни між двома коміттами 
git log --oneline --decorate --graph --all
git revert -- відкрутити назад чот таке 

================Гілки===============================
git branch "назва" -- створення нової гілки 
git checkout "назва" -- перехід між гілками
git branch -d "назва" -- видалення гілки
git merge -- злиття коммітів або перенесення покажчика з минулого коміта на зарішній


================віддалені сховища===================
git remote add <name> <url> -- додаємо віддалене сховище з назвою name і по силці url
git remote remove <name> -- видалити силку на віддаленек сховище 
git clone -- копіювання файлів звідкись
git pull -- витягнути з віддаленого сховища всі нові коміти тіпа
git push -- закинути на віддалене сховище всі комміти 

===================вивести чиото===================
git log --graph --pretty=oneline --abbrev-commit  
git log --oneline --decorate --graph --all
